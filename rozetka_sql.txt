
Запросы  выполнялись в PostgreSQL
-- 1
SELECT EMPID, EMPNAME
FROM EMP
LIMIT 3;

-- 2
SELECT SALARY
FROM EMP
ORDER BY SALARY DESC
LIMIT 2;

-- 3
SELECT EMP.EMPNAME, EMP.DOB, DEPT.DEPTNAME
FROM EMP
JOIN DEPT ON EMP.deptid = DEPT.deptid
WHERE DEPT.DEPTNAME = 'QA'
ORDER BY EMP.EMPNAME;

--4
SELECT EMP.EMPNAME, EXTRACT(year FROM AGE(CURRENT_DATE, EMP.DOB)) AGE, DEPT.DEPTNAME
FROM EMP
JOIN DEPT ON EMP.deptid = DEPT.deptid
ORDER BY AGE DESC;

--5
SELECT DEPT.DEPTNAME, COUNT(EMP.EMPID) EMPCOUNT
FROM DEPT
LEFT JOIN EMP ON DEPT.deptid = EMP.deptid
GROUP BY DEPT.DEPTNAME
ORDER BY EMPCOUNT;

--6
SELECT EMP.EMPNAME, MANAG.EMPNAME MANAGNAME, DEPT.DEPTNAME
FROM EMP
LEFT JOIN EMP MANAG ON EMP.MGRID = MANAG.EMPID
JOIN DEPT ON EMP.DEPTID = DEPT.DEPTID;
